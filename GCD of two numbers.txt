function gcd()
{
var x,y;
x=parseInt(document.f1.t1.value);
y=parseInt(document.f1.t2.value);
while(x!=y)
{
if(x>y)
x=x-y;
else
y=y-x;
}
document.f1.t3.value=x;
}
